### Method 2: API Script (Fully Automated) ‚úÖ RECOMMENDED

```javascript
// launch-20-agents.js
import Anthropic from '@anthropic-ai/sdk';
import fs from 'fs';

const client = new Anthropic({
  apiKey: process.env.ANTHROPIC_API_KEY,
});

// Load all 20 task prompts
const tasks = [];
for (let i = 1; i <= 20; i++) {
  tasks.push({
    id: i,
    name: `Task ${i}`,
    prompt: fs.readFileSync(`./prompts/task-${i}.txt`, 'utf8'),
  });
}

console.log('Launching 20 autonomous agents in parallel...');

// Launch all 20 agents simultaneously
const agents = tasks.map(async (task) => {
  console.log(`Starting Task ${task.id}: ${task.name}`);

  const message = await client.messages.create({
    model: 'claude-opus-4-20250514', // Most expensive model
    max_tokens: 16000, // Maximum output
    messages: [{
      role: 'user',
      content: task.prompt,
    }],
    // Enable extended thinking for deep analysis
    thinking: {
      type: 'enabled',
      budget_tokens: 32000, // Maximum thinking
    },
  });

  console.log(`‚úÖ Task ${task.id} completed!`);

  return {
    taskId: task.id,
    result: message.content,
    usage: message.usage,
  };
});

// Wait for all agents to complete
const results = await Promise.all(agents);

// Generate summary report
console.log('\n=== ALL 20 AGENTS COMPLETED ===\n');

let totalCost = 0;
results.forEach((result) => {
  const inputCost = (result.usage.input_tokens / 1_000_000) * 15;
  const outputCost = (result.usage.output_tokens / 1_000_000) * 75;
  const cost = inputCost + outputCost;
  totalCost += cost;

  console.log(`Task ${result.taskId}:`);
  console.log(`  Input tokens: ${result.usage.input_tokens}`);
  console.log(`  Output tokens: ${result.usage.output_tokens}`);
  console.log(`  Cost: $${cost.toFixed(2)}`);
  console.log('');
});

console.log(`\nTOTAL COST: $${totalCost.toFixed(2)}`);
console.log(`\nCheck GitHub for 20 new commits!`);

// Save results
fs.writeFileSync(
  'agent-results.json',
  JSON.stringify(results, null, 2)
);
```
